# 店铺表

```mysql
drop table if exists sys_bsi_shop;
create table sys_bsi_shop (
  shop_id           int(11)         auto_increment    comment '店铺编号',
  shop_del_flag     char(1)         default 'N'       comment '删除标志（Y删除 N未删除）',
  shop_type         char(1)         default '0'       comment '店铺类型(0超市，1食品，)',
  shop_status       char(1)         default '0'       comment '店铺状态（0正常 1休息 2停业）',
  shop_tel          varchar(50)     default ''        comment '店铺联系方式',
  shop_address      varchar(50)     default ''        comment '店铺地址',
  shop_banking      varchar(50)     default ''        comment '店铺营业时间',
  shop_name         varchar(100)    default ''        comment '店铺名称',
  shop_introduce    varchar(200)    default ''        comment '店铺信息介绍',
  shop_item         varchar(200)    default ''        comment '店铺特色',
  shop_discount     varchar(200)    default ''        comment '店铺打折优惠信息',
  shop_open_time    datetime                         comment '店铺开业时间',
  primary key (shop_id)
) engine=innodb auto_increment=1 comment = '店铺信息表';
```
